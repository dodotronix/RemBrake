  <Script ScriptDone="true"># !!! Do not forget to set the Servo 0 as Input

goto main_loop

sub remote_button
begin
dup
while
swap
0 get_position 500 greater_than
1 delay
plus
swap
1 minus
repeat
drop
return

sub button_active
begin
0 10 remote_button
10 equals
dup
rot
greater_than
if return endif
repeat

sub move_servo
1 get_position 0 equals
if
8000 1 servo
else
0 1 servo
endif
return

main_loop:
1 # button status
begin
button_active
move_servo
led_on
100 delay
led_off
100 delay
repeat</Script>
